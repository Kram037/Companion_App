rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /Campagne/{campagnaId} {
      // Allow read/write if user is authenticated
      // The app filters by userId on client side for security
      allow read, write: if request.auth != null;
    }
    
    match /Utenti/{docId} {
      // Users can only read/write their own document
      // Document ID format: utente_<cid>_<nome>
      // We check the uid field in the document data
      allow read: if request.auth != null && 
                     (resource.data.uid == request.auth.uid || 
                      request.query.limit <= 1); // Allow queries with limit 1
      allow create: if request.auth != null && 
                       request.resource.data.uid == request.auth.uid;
      allow update, delete: if request.auth != null && 
                                resource.data.uid == request.auth.uid;
    }
  }
}
